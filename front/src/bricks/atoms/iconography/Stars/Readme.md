### Import
```js static
import { Stars } from 'pretto-ui'
```

### Usage
```html
<Stars />

```
### Examples
#### Basic:
```jsx
<Stars />
```

#### 5 Stars:
```jsx
<Stars number={5} />
```

#### Sizes:
```jsx
<div style={{display: 'inline-flex', alignItems: 'center', justifyContent: 'space-between', width: 100}}>
  <Stars size="small" />
  <Stars size="medium" />
  <Stars size="large" />
</div>
```
