### Import
```js static
import { Content } from 'pretto-ui'
```

### Usage
```html
<Content>
  <div>Any content</div>
</Content>
```

### Examples
#### Basic:
```jsx
<Card>
  <Content>What a beautiful card</Content>
</Card>
```

#### Padding: small
```jsx
<Card>
  <Content size="small">What a beautiful card</Content>
</Card>
```
