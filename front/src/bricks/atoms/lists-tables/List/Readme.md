### Import
```js static
import { List, ListItem } from 'pretto-ui'
```

### Usage
```html
<List>
  <ListItem>Content Item List</ListItem>
  <ListItem>Content Item List</ListItem>
</List>
```

### Examples
#### Basic
```jsx
<List>
  <ListItem>Hello</ListItem>
  <ListItem>World</ListItem>
</List>
```

#### Format: documents
```jsx
<div style={{background: 'orange'}}>
  <List format="documents">
    <ListItem>Hello</ListItem>
    <ListItem>World</ListItem>
  </List>
</div>
```
